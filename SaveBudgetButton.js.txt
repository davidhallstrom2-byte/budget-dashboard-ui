// src/components/ui/SaveBudgetButton.jsx
import { useState } from 'react';
import { Save, Download, Check, AlertCircle } from 'lucide-react';
import { useBudgetState } from '../../utils/state';

export default function SaveBudgetButton() {
  const [saved, setSaved] = useState(false);
  const [showInstructions, setShowInstructions] = useState(false);
  const exportToJson = useBudgetState((state) => state.exportToJson);

  const handleSave = () => {
    try {
      // Get current state as JSON
      const data = exportToJson();
      const jsonString = JSON.stringify(data, null, 2);
      
      // Create blob and download
      const blob = new Blob([jsonString], { type: 'application/json' });
      const url = URL.createObjectURL(blob);
      const link = document.createElement('a');
      link.href = url;
      link.download = 'budget-data.json';
      document.body.appendChild(link);
      link.click();
      document.body.removeChild(link);
      URL.revokeObjectURL(url);
      
      // Show success state
      setSaved(true);
      setShowInstructions(true);
      
      // Reset after 3 seconds
      setTimeout(() => {
        setSaved(false);
      }, 3000);
      
    } catch (error) {
      console.error('Save failed:', error);
      alert('Failed to save budget: ' + error.message);
    }
  };

  return (
    <div className="relative">
      <button
        onClick={handleSave}
        className={`flex items-center gap-2 px-4 py-2 rounded-lg font-medium transition-all ${
          saved
            ? 'bg-green-500 text-white'
            : 'bg-blue-500 hover:bg-blue-600 text-white'
        }`}
      >
        {saved ? (
          <>
            <Check size={18} />
            <span>Saved!</span>
          </>
        ) : (
          <>
            <Save size={18} />
            <span>Save Budget</span>
          </>
        )}
      </button>

      {showInstructions && (
        <div className="absolute top-full mt-2 right-0 w-96 bg-blue-50 border border-blue-200 rounded-lg p-4 shadow-lg z-50">
          <div className="flex items-start gap-2">
            <AlertCircle className="text-blue-600 flex-shrink-0 mt-0.5" size={20} />
            <div>
              <h4 className="font-semibold text-blue-900 mb-2">File Downloaded</h4>
              <div className="text-sm text-blue-800 space-y-2">
                <p>Your budget has been downloaded as <code className="bg-blue-100 px-1 rounded">budget-data.json</code></p>
                <p className="font-medium">To make changes permanent:</p>
                <ol className="list-decimal list-inside space-y-1 text-xs">
                  <li>Find the downloaded file in your Downloads folder</li>
                  <li>Copy it to: <code className="bg-blue-100 px-1 rounded text-xs break-all">C:\Users\david\Local Sites\main-dashboard\app\public\budget-dashboard-fs\ui\public\restore\</code></li>
                  <li>Replace the existing <code className="bg-blue-100 px-1 rounded">budget-data.json</code></li>
                  <li>Refresh this page</li>
                </ol>
              </div>
              <button
                onClick={() => setShowInstructions(false)}
                className="mt-3 text-xs text-blue-600 hover:text-blue-700 underline"
              >
                Got it, close
              </button>
            </div>
          </div>
        </div>
      )}
    </div>
  );
}
